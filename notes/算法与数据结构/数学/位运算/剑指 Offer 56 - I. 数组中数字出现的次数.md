class Solution:
    def singleNumbers(self, nums: List[int]) -> List[int]:
        mix = 0
        for num in nums:
            mix ^= num
        mask = 1
        # fixme: 在这一位上两者的异或值为1，可以用来区分ab
        while (mix & mask) == 0:
            mask <<= 1
        a, b = 0, 0
        for num in nums:
            if num & mask:
                a ^= num
            else:
                b ^= num
        return [a, b]
