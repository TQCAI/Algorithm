```python
class Solution:

        p = head
        pp = None
        while p:
            next = p.next
            p.next = pp
            pp = p
            p = next
        return pp
```

```python
class Solution:
    def reverseBetween(self, head: ListNode, m: int, n: int) -> ListNode:
        p = head
        # dummy结点
        res = ListNode(0)
        p_res = res
        for _ in range(m - 1):
            # 添加到结果链表中
            p_res.next = p
            p_res = p_res.next
            p = p.next
        # 此时的res： 1->∅
        # p 的 previous
        pp = None
        idx = 0
        # 此时的 p2：2->...
        p2 = p
        while p and idx < (n - m + 1):
            # ∅    2 -> 3 -> 4
            # pp   p    next
            # ∅ <- 2  3 -> 4
            #     pp  p   next
            idx += 1
            next = p.next
            p.next = pp
            pp = p
            p = next
        # 此时的p：5 -> ∅ 
        if p2:
            # 2 -> 5 -> ∅ 
            p2.next = p
        # 1 -> 4 -> 3 -> 2 -> 5 -> ∅ 
        p_res.next = pp
        return res.next
```

```python
from structure import ListNode


class Solution:
    def recurse(self, head: ListNode):
        #                     写错↓
        if head is None or head.next is None:
            return head  # ←写错
        last = self.recurse(head.next)
        head.next.next = head
        head.next = None
        return last

    successor = None

    def recurseN(self, head: ListNode, n: int):
        if n == 1:
            self.successor = head.next
            return head
        last = self.recurseN(head.next, n - 1)
        head.next.next = head
        head.next = self.successor
        return last

    def reverseBetween(self, head: ListNode, m: int, n: int) -> ListNode:
        if m == 1:
            return self.recurseN(head, n)
        head.next = self.reverseBetween(head.next, m - 1, n - 1)
        return head


node = ListNode.fromList([1, 2, 3, 4, 5])
print(Solution().recurse(node))
node = ListNode.fromList([1, 2, 3, 4, 5])
print(Solution().recurseN(node, 2))
node = ListNode.fromList([1, 2, 3, 4, 5])
print(Solution().reverseBetween(node, 2, 4))
```
