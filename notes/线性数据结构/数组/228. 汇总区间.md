class Solution:
    def summaryRanges(self, nums: List[int]) -> List[str]:
        if not nums:
            return []
        res=[]
        s=0
        e=0
        for i in range(1, len(nums)):
            if nums[i]-nums[i-1]==1:
                e=i
            else:
                res.append(str(nums[s]) if s==e else f"{nums[s]}->{nums[e]}")
                s=e=i
        res.append(str(nums[s]) if s==e else f"{nums[s]}->{nums[e]}")
        return res
